509  void destroy(const entity_type entity) {
510  assert(valid(entity));
511 
512  for(auto pos = pools.size(); pos; --pos) {
513  auto &tup = pools[pos-1];
514  auto &cpool = std::get<0>(tup);
515 
516  if(cpool && cpool->has(entity)) {
517  std::get<2>(tup).publish(*this, entity);
