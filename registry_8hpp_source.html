512  if(const auto req = (to_integral(hint) & traits_type::entity_mask); !(req < entities.size())) {

1415  constexpr auto size = sizeof...(Owned) + sizeof...(Get) + sizeof...(Exclude);

1419  return gdata.size == size

1421  && (gdata.get(type_info<std::decay_t<Get>>::id()) && ...)
1422  && (gdata.exclude(type_info<Exclude>::id()) && ...);
1423  }); it != groups.cend())

1425  handler = static_cast<handler_type *>(it->group.get());
1426  }
1427

1429  group_data candidate = {
1430  size,
1431  { new handler_type{}, [](void *instance) { delete static_cast<handler_type *>(instance); } },
1432  []([[maybe_unused]] const id_type ctype) ENTT_NOEXCEPT { return ((ctype == type_info<std::decay_t<Owned>>::id()) || ...); },
1433  []([[maybe_unused]] const id_type ctype) ENTT_NOEXCEPT { return ((ctype == type_info<std::decay_t<Get>>::id()) || ...); },
1434  []([[maybe_unused]] const id_type ctype) ENTT_NOEXCEPT { return ((ctype == type_info<Exclude>::id()) || ...); },
1435  };
1436

1440  const void *discard_if = nullptr;
1441 
1442  if constexpr(sizeof...(Owned) == 0) {
1443  groups.push_back(std::move(candidate));
1444  } else {
1445  ENTT_ASSERT(std::all_of(groups.cbegin(), groups.cend(), [size](const auto &gdata) {
1446  const auto overlapping = (0u + ... + gdata.owned(type_info<std::decay_t<Owned>>::id()));
1447  const auto sz = overlapping + (0u + ... + gdata.get(type_info<std::decay_t<Get>>::id())) + (0u + ... + gdata.exclude(type_info<Exclude>::id()));
1448  return !overlapping || ((sz == size) || (sz == gdata.size));
1449  }));
1450

1452  return !(0u + ... + gdata.owned(type_info<std::decay_t<Owned>>::id())) || (size > (gdata.size));
1453  });

1456  return (0u + ... + gdata.owned(type_info<std::decay_t<Owned>>::id()));
1457  });
1458 
1459  maybe_valid_if = (next == groups.cend() ? maybe_valid_if : next->group.get());
1460  discard_if = (prev == groups.crend() ? discard_if : prev->group.get());
1461  groups.insert(next, std::move(candidate));
1462  }

1465
