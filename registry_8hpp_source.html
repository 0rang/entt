582  void destroy(const entity_type entity) {
583  assert(valid(entity));
584 
585  for(auto pos = pools.size(); pos; --pos) {
586  auto &cpool = pools[pos-1];
587 
588  if(cpool && cpool->has(entity)) {
589  cpool->destroy(entity);
590  }
591  };
592 
593  for(auto pos = tags.size(); pos; --pos) {
594  auto &tag = tags[pos-1];
595 
596  if(tag && tag->get() == entity) {
597  tag->destroy();
598  }
599  };
600 
601  // just a way to protect users from listeners that attach components
602  assert(orphan(entity));
603 
604  // lengthens the implicit list of destroyed entities
605  const auto entt = entity & traits_type::entity_mask;
606  const auto version = ((entity >> traits_type::entity_shift) + 1) << traits_type::entity_shift;
607  const auto node = (available ? next : ((entt + 1) & traits_type::entity_mask)) | version;
608  entities[entt] = node;
609  next = entt;
610  ++available;
611  }
