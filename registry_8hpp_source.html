249  template<typename Component>
250  pool_type<Component> * assure() {
251  const auto ctype = to_integer(type<Component>());
252  pool_data *pdata = nullptr;
253 
254  if constexpr(is_named_type_v<Component>) {
255  const auto it = std::find_if(pools.begin()+skip_family_pools, pools.end(), [ctype](const auto &candidate) {
256  return candidate.runtime_type == ctype;
257  });
258 
259  pdata = (it == pools.cend() ? &pools.emplace_back() : &(*it));
260  } else {
261  if(!(ctype < skip_family_pools)) {
262  pools.reserve(pools.size()+ctype-skip_family_pools+1);
263 
264  while(!(ctype < skip_family_pools)) {
265  pools.emplace(pools.begin()+(skip_family_pools++), pool_data{});
266  }
267  }
268 
269  pdata = &pools[ctype];
270  }
271 
272  if(!pdata->pool) {
273  pdata->runtime_type = ctype;
274  pdata->pool = std::make_unique<pool_type<Component>>();
275 
276  pdata->remove = [](sparse_set<Entity> &cpool, basic_registry &registry, const Entity entt) {
277  static_cast<pool_type<Component> &>(cpool).remove(registry, entt);
278  };
279 
280  if constexpr(std::is_copy_constructible_v<std::decay_t<Component>>) {
281  pdata->clone = [](const sparse_set<Entity> &cpool) -> std::unique_ptr<sparse_set<Entity>> {
282  return std::make_unique<pool_type<Component>>(static_cast<const pool_type<Component> &>(cpool));
283  };
284 
285  pdata->stomp = [](const sparse_set<Entity> &cpool, const Entity from, basic_registry &other, const Entity to) {
286  other.assign_or_replace<Component>(to, static_cast<const pool_type<Component> &>(cpool).get(from));
287  };
288  } else {
289  pdata->clone = nullptr;
290  pdata->stomp = nullptr;
291  }
292  }
293 
294  return static_cast<pool_type<Component> *>(pdata->pool.get());
295  }

588  static_assert(std::is_convertible_v<entity_type, typename std::iterator_traits<It>::value_type>);
589
