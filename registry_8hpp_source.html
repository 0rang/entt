501  [[nodiscard]] entity_type create(const entity_type hint) {
502  ENTT_ASSERT(hint != null);
503  entity_type entt;
504 
505  if(const auto req = (to_integral(hint) & traits_type::entity_mask); !(req < entities.size())) {
506  entities.reserve(req + 1);
507 
508  for(auto pos = entities.size(); pos < req; ++pos) {
509  entities.emplace_back(destroyed);
510  destroyed = entity_type{static_cast<typename traits_type::entity_type>(pos)};
511  }
512 
513  entt = entities.emplace_back(hint);
514  } else if(const auto curr = (to_integral(entities[req]) & traits_type::entity_mask); req == curr) {
515  entt = create();
516  } else {

519  *it = entity_type{curr | (to_integral(*it) & (traits_type::version_mask << traits_type::entity_shift))};
520  entt = entities[req] = hint;
521  }

1302  group_data candidate = {
1303  size,
1304  { new handler_type{}, [](void *instance) { delete static_cast<handler_type *>(instance); } },
1305  []([[maybe_unused]] const id_type ctype) ENTT_NOEXCEPT { return ((ctype == type_info<std::decay_t<Owned>>::id()) || ...); },

1313  const void *discard_if = nullptr;
1314

1316  groups.push_back(std::move(candidate));
1317  } else {
1318  ENTT_ASSERT(std::all_of(groups.cbegin(), groups.cend(), [size](const auto &gdata) {
1319  const auto overlapping = (0u + ... + gdata.owned(type_info<std::decay_t<Owned>>::id()));

1321  return !overlapping || ((sz == size) || (sz == gdata.size));
1322  }));
1323

1325  return !(0u + ... + gdata.owned(type_info<std::decay_t<Owned>>::id())) || (size > gdata.size);
1326  });
1327 
1328  const auto prev = std::find_if(std::make_reverse_iterator(next), groups.crend(), [](const auto &gdata) {
1329  return (0u + ... + gdata.owned(type_info<std::decay_t<Owned>>::id()));
1330  });
1331 
1332  maybe_valid_if = (next == groups.cend() ? maybe_valid_if : next->group.get());
1333  discard_if = (prev == groups.crend() ? discard_if : prev->group.get());
1334  groups.insert(next, std::move(candidate));
1335  }

1338  (on_construct<std::decay_t<Get>>().before(maybe_valid_if).template connect<&handler_type::template maybe_valid_if<std::decay_t<Get>>>(*handler), ...);

1343  (on_construct<Exclude>().before(discard_if).template connect<&handler_type::discard_if>(*handler), ...);
