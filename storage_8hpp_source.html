485  template<typename Compare, typename Sort = std_sort, typename... Args>
486  void sort(iterator first, iterator last, Compare compare, Sort algo = Sort{}, Args &&... args) {
487  ENTT_ASSERT(!(last < first));
488  ENTT_ASSERT(!(last > end()));
489 
490  const auto from = underlying_type::begin() + std::distance(begin(), first);
491  const auto to = from + std::distance(first, last);
492 
493  const auto apply = [this](const auto lhs, const auto rhs) {
494  std::swap(instances[underlying_type::index(lhs)], instances[underlying_type::index(rhs)]);
495  };
496 
497  if constexpr(std::is_invocable_v<Compare, const object_type &, const object_type &>) {
498  underlying_type::arrange(from, to, std::move(apply), [this, compare = std::move(compare)](const auto lhs, const auto rhs) {
499  return compare(std::as_const(instances[underlying_type::index(lhs)]), std::as_const(instances[underlying_type::index(rhs)]));
500  }, std::move(algo), std::forward<Args>(args)...);
501  } else {
502  underlying_type::arrange(from, to, std::move(apply), std::move(compare), std::move(algo), std::forward<Args>(args)...);
503  }
504  }
