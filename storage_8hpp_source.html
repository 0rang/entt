421  void sort(Compare compare, Sort algo = Sort{}, Args &&... args) {
422  std::vector<size_type> copy(instances.size());
423  std::iota(copy.begin(), copy.end(), 0);
424 
425  if constexpr(std::is_invocable_v<Compare, const object_type &, const object_type &>) {
426  static_assert(!std::is_empty_v<object_type>);
427 
428  algo(copy.rbegin(), copy.rend(), [this, compare = std::move(compare)](const auto lhs, const auto rhs) {
429  return compare(std::as_const(instances[lhs]), std::as_const(instances[rhs]));
430  }, std::forward<Args>(args)...);
431  } else {
432  algo(copy.rbegin(), copy.rend(), [compare = std::move(compare), data = underlying_type::data()](const auto lhs, const auto rhs) {
433  return compare(data[lhs], data[rhs]);
434  }, std::forward<Args>(args)...);
435  }
436 
437  for(size_type pos = 0, last = copy.size(); pos < last; ++pos) {
438  auto curr = pos;
439  auto next = copy[curr];
440 
441  while(curr != next) {
442  const auto lhs = copy[curr];
443  const auto rhs = copy[next];
444  std::swap(instances[lhs], instances[rhs]);
445  underlying_type::swap(lhs, rhs);
446  copy[curr] = curr;
447  curr = next;
448  next = copy[curr];
449  }
450  }
