513  template<typename Func, typename... Strong, typename... Weak>
514  void traverse(Func func, type_list<Strong...>, type_list<Weak...>) const {
515  [[maybe_unused]] auto it = std::make_tuple((std::get<pool_type<Strong> *>(pools)->end() - *length)...);
516  [[maybe_unused]] auto data = std::get<0>(pools)->sparse_set<entity_type>::end() - *length;
517 
518  for(auto next = *length; next; --next) {
519  if constexpr(std::is_invocable_v<Func, decltype(get<Strong>({}))..., decltype(get<Weak>({}))...>) {
520  if constexpr(sizeof...(Weak) == 0) {
521  func(*(std::get<component_iterator_type<Strong>>(it)++)...);
522  } else {
523  const auto entt = *(data++);
524  func(*(std::get<component_iterator_type<Strong>>(it)++)..., std::get<pool_type<Weak> *>(pools)->get(entt)...);
525  }
526  } else {
527  const auto entt = *(data++);
528  func(entt, *(std::get<component_iterator_type<Strong>>(it)++)..., std::get<pool_type<Weak> *>(pools)->get(entt)...);
529  }
530  }
