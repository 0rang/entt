506  template<typename Func, typename... Strong, typename... Weak>
507  void traverse(Func func, type_list<Strong...>, type_list<Weak...>) const {
508  [[maybe_unused]] auto raw = std::make_tuple((std::get<pool_type<Strong> *>(pools)->end() - *length)...);
509  [[maybe_unused]] auto data = std::get<pool_type<Owned> *>(pools)->sparse_set<entity_type>::end() - *length;
510 
511  for(auto next = *length; next; --next) {
512  if constexpr(std::is_invocable_v<Func, decltype(get<Strong>({}))..., decltype(get<Weak>({}))...>) {
513  if constexpr(sizeof...(Weak) == 0) {
514  func(*(std::get<component_iterator_type<Strong>>(raw)++)...);
515  } else {
516  const auto entt = *(data++);
517  func(*(std::get<component_iterator_type<Strong>>(raw)++)..., std::get<pool_type<Weak> *>(pools)->get(entt)...);
518  }
519  } else {
520  const auto entt = *(data++);
521  func(entt, *(std::get<component_iterator_type<Strong>>(raw)++)..., std::get<pool_type<Weak> *>(pools)->get(entt)...);
522  }
523  }

829  auto *cpool = std::get<pool_type<Owned> *>(pools);
830 
831  if constexpr(sizeof...(Component) == 0) {
832  static_assert(std::is_invocable_v<Compare, const entity_type, const entity_type>);
833  cpool->sort(cpool->end()-*length, cpool->end(), std::move(compare), std::move(algo), std::forward<Args>(args)...);
834  } else {

837  return compare(this->get<Component...>(lhs), this->get<Component...>(rhs));

839  }

843  const auto pos = cpool->index(curr);

845  if(pos != next) {
