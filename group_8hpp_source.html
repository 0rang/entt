500  template<typename Func, typename... Strong, typename... Weak>
501  void traverse(Func func, type_list<Strong...>, type_list<Weak...>) const {
502  [[maybe_unused]] auto raw = std::make_tuple((std::get<pool_type<Strong> *>(pools)->end() - *length)...);
503  [[maybe_unused]] auto data = std::get<0>(pools)->sparse_set<entity_type>::end() - *length;
504 
505  for(auto next = *length; next; --next) {
506  if constexpr(std::is_invocable_v<Func, decltype(get<Strong>({}))..., decltype(get<Weak>({}))...>) {
507  if constexpr(sizeof...(Weak) == 0) {
508  func(*(std::get<component_iterator_type<Strong>>(raw)++)...);
509  } else {
510  const auto entt = *(data++);
511  func(*(std::get<component_iterator_type<Strong>>(raw)++)..., std::get<pool_type<Weak> *>(pools)->get(entt)...);
512  }
513  } else {
514  const auto entt = *(data++);
515  func(entt, *(std::get<component_iterator_type<Strong>>(raw)++)..., std::get<pool_type<Weak> *>(pools)->get(entt)...);
516  }
517  }
