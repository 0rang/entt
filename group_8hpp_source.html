517  template<typename Func, typename... Strong, typename... Weak>
518  void traverse(Func func, type_list<Strong...>, type_list<Weak...>) const {
519  [[maybe_unused]] auto it = std::make_tuple((std::get<pool_type<Strong> *>(pools)->end() - *length)...);
520  [[maybe_unused]] auto data = std::get<0>(pools)->sparse_set<entity_type>::end() - *length;
521 
522  for(auto next = *length; next; --next) {
523  if constexpr(std::is_invocable_v<Func, decltype(get<Strong>({}))..., decltype(get<Weak>({}))...>) {
524  if constexpr(sizeof...(Weak) == 0) {
525  func(*(std::get<component_iterator<Strong>>(it)++)...);
526  } else {
527  const auto entt = *(data++);
528  func(*(std::get<component_iterator<Strong>>(it)++)..., std::get<pool_type<Weak> *>(pools)->get(entt)...);
529  }
530  } else {
531  const auto entt = *(data++);
532  func(entt, *(std::get<component_iterator<Strong>>(it)++)..., std::get<pool_type<Weak> *>(pools)->get(entt)...);
533  }
534  }
