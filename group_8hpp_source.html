740  void sort(Compare compare, Sort algo = Sort{}, Args &&... args) {
741  std::vector<size_type> copy(*length);
742  std::iota(copy.begin(), copy.end(), 0);
743 
744  if constexpr(sizeof...(Component) == 0) {
745  algo(copy.rbegin(), copy.rend(), [compare = std::move(compare), data = data()](const auto lhs, const auto rhs) {
746  return compare(data[lhs], data[rhs]);
747  }, std::forward<Args>(args)...);
748  } else {
749  algo(copy.rbegin(), copy.rend(), [compare = std::move(compare), this](const auto lhs, const auto rhs) {
750  // useless this-> used to suppress a warning with clang
751  return compare(this->from_index<Component>(lhs)..., this->from_index<Component>(rhs)...);
752  }, std::forward<Args>(args)...);
753  }
754 
755  for(size_type pos = 0, last = copy.size(); pos < last; ++pos) {
756  auto curr = pos;
757  auto next = copy[curr];
758 
759  while(curr != next) {
760  const auto lhs = copy[curr];
761  const auto rhs = copy[next];
762  (swap<Owned>(0, std::get<pool_type<Owned> *>(pools), lhs, rhs), ...);
763  copy[curr] = curr;
764  curr = next;
765  next = copy[curr];
766  }
767  }
768  }
