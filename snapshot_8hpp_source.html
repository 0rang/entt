327  template<typename Container>
328  auto update(int, Container &container)
329  -> decltype(typename Container::mapped_type{}, void()) {
330  // map like container
331  Container other;
332 
333  for(auto &&pair: container) {
334  using first_type = std::remove_const_t<typename std::decay_t<decltype(pair)>::first_type>;
335  using second_type = typename std::decay_t<decltype(pair)>::second_type;
336 
337  if constexpr(std::is_same_v<first_type, entity_type> && std::is_same_v<second_type, entity_type>) {
338  other.emplace(map(pair.first), map(pair.second));
339  } else if constexpr(std::is_same_v<first_type, entity_type>) {
340  other.emplace(map(pair.first), std::move(pair.second));
341  } else {
342  static_assert(std::is_same_v<second_type, entity_type>, "Neither the key nor the value are of entity type");
343  other.emplace(std::move(pair.first), map(pair.second));
344  }
345  }
346 
347  std::swap(container, other);
348  }
