342  template<typename Container>
343  auto update(int, Container &container)
344  -> decltype(typename Container::mapped_type{}, void()) {
345  // map like container
346  Container other;
347 
348  for(auto &&pair: container) {
349  using first_type = std::remove_const_t<typename std::decay_t<decltype(pair)>::first_type>;
350  using second_type = typename std::decay_t<decltype(pair)>::second_type;
351 
352  if constexpr(std::is_same_v<first_type, entity_type> && std::is_same_v<second_type, entity_type>) {
353  other.emplace(map(pair.first), map(pair.second));
354  } else if constexpr(std::is_same_v<first_type, entity_type>) {
355  other.emplace(map(pair.first), std::move(pair.second));
356  } else {
357  static_assert(std::is_same_v<second_type, entity_type>);
358  other.emplace(std::move(pair.first), map(pair.second));
359  }
360  }
361 
362  std::swap(container, other);
363  }
