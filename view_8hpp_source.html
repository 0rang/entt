162  template<typename Comp, typename Func, typename... Other, typename... Type>
163  void traverse(Func func, type_list<Other...>, type_list<Type...>) const {
164  const auto end = std::get<pool_type<Comp> *>(pools)->sparse_set<Entity>::end();
165  auto begin = std::get<pool_type<Comp> *>(pools)->sparse_set<Entity>::begin();
166 
167  if constexpr(std::disjunction_v<std::is_same<Comp, Type>...>) {
168  std::for_each(begin, end, [this, raw = std::get<pool_type<Comp> *>(pools)->begin(), &func](const auto entity) mutable {
169  auto curr = raw++;
170 
171  if((std::get<pool_type<Other> *>(pools)->has(entity) && ...)) {
172  if constexpr(std::is_invocable_v<Func, decltype(get<Type>({}))...>) {
173  func(get<Comp, Type>(curr, std::get<pool_type<Type> *>(pools), entity)...);
174  } else {
175  func(entity, get<Comp, Type>(curr, std::get<pool_type<Type> *>(pools), entity)...);
176  }
177  }
178  });
179  } else {
180  std::for_each(begin, end, [this, &func](const auto entity) mutable {
181  if((std::get<pool_type<Other> *>(pools)->has(entity) && ...)) {
182  if constexpr(std::is_invocable_v<Func, decltype(get<Type>({}))...>) {
183  func(std::get<pool_type<Type> *>(pools)->get(entity)...);
184  } else {
185  func(entity, std::get<pool_type<Type> *>(pools)->get(entity)...);
