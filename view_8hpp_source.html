181  if constexpr(std::disjunction_v<std::is_same<Comp, Type>...>) {
182  auto it = std::get<pool_type<Comp> *>(pools)->begin();
183 
184  for(const auto entt: static_cast<const sparse_set<entity_type> &>(*std::get<pool_type<Comp> *>(pools))) {
185  auto curr = it++;
186 
187  if(((std::is_same_v<Comp, Component> || std::get<pool_type<Component> *>(pools)->has(entt)) && ...) && (!std::get<pool_type<Exclude> *>(pools)->has(entt) && ...)) {
188  if constexpr(std::is_invocable_v<Func, decltype(get<Type>({}))...>) {
189  func(get<Comp, Type>(curr, std::get<pool_type<Type> *>(pools), entt)...);
190  } else {

192  }
193  }
194  }
195  } else {

197  if(((std::is_same_v<Comp, Component> || std::get<pool_type<Component> *>(pools)->has(entt)) && ...) && (!std::get<pool_type<Exclude> *>(pools)->has(entt) && ...)) {

199  func(std::get<pool_type<Type> *>(pools)->get(entt)...);
