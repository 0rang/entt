173  template<typename Comp, typename Func, typename... Other, typename... Type>
174  void traverse(Func func, type_list<Other...>, type_list<Type...>) const {
175  const auto end = std::get<pool_type<Comp> *>(pools)->sparse_set<Entity>::end();
176  auto begin = std::get<pool_type<Comp> *>(pools)->sparse_set<Entity>::begin();
177 
178  if constexpr(std::disjunction_v<std::is_same<Comp, Type>...>) {
179  std::for_each(begin, end, [this, raw = std::get<pool_type<Comp> *>(pools)->begin(), &func](const auto entity) mutable {
180  auto curr = raw++;
181 
182  if((std::get<pool_type<Other> *>(pools)->has(entity) && ...) && (!std::get<pool_type<Exclude> *>(filter)->has(entity) && ...)) {
183  if constexpr(std::is_invocable_v<Func, decltype(get<Type>({}))...>) {
184  func(get<Comp, Type>(curr, std::get<pool_type<Type> *>(pools), entity)...);
185  } else {
186  func(entity, get<Comp, Type>(curr, std::get<pool_type<Type> *>(pools), entity)...);
187  }
188  }
189  });
190  } else {
191  std::for_each(begin, end, [this, &func](const auto entity) {
192  if((std::get<pool_type<Other> *>(pools)->has(entity) && ...) && (!std::get<pool_type<Exclude> *>(filter)->has(entity) && ...)) {
193  if constexpr(std::is_invocable_v<Func, decltype(get<Type>({}))...>) {
194  func(std::get<pool_type<Type> *>(pools)->get(entity)...);
195  } else {
196  func(entity, std::get<pool_type<Type> *>(pools)->get(entity)...);
197  }
198  }
199  });
200  }
