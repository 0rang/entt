578  template<typename Comp, typename Func, std::size_t... Indexes>
579  void each(const pool_type<Comp> &cpool, Func func, std::index_sequence<Indexes...>) const {
580  const auto other = unchecked(&cpool);
581  std::array<underlying_iterator_type, sizeof...(Component)> data{{cpool.view_type::cbegin(), std::get<Indexes>(other)->cbegin()...}};
582  auto raw = std::make_tuple(pool<Component>().cbegin()...);
583  const auto end = cpool.view_type::cend();
584  std::size_t pos{};
585 
586  // we can directly use the raw iterators if pools are ordered
587  while(!pos && data[0] != end) {
588  for(pos = data.size() - 1; pos && *(data[pos]++) == *data[pos-1]; --pos);
589 
590  if(!pos) {
591  func(*(data[0]++), *(std::get<component_iterator_type<Component>>(raw)++)...);
592  }
593  }
594 
595  auto it = std::get<component_iterator_type<Comp>>(raw);
596  const auto ext = extent();
597 
598  // fallback to visit what remains using indirections
599  for(; data[0] != end; ++data[0], ++it) {
600  const auto entity = *data[0];
601  const auto sz = size_type(entity & traits_type::entity_mask);
602 
603  if(sz < ext && std::all_of(other.cbegin(), other.cend(), [entity](const view_type *view) { return view->fast(entity); })) {
604  // avoided at least the indirection due to the sparse set for the pivot type (see get for more details)
605  func(entity, get<Comp, Component>(it, entity)...);
606  }
607  }
608  }
