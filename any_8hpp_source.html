44  template<typename Type>
45  static const void * basic_vtable([[maybe_unused]] const operation op, [[maybe_unused]] const any &from, [[maybe_unused]] const void *to) {
46  if constexpr(!std::is_void_v<Type>) {
47  if constexpr(std::is_lvalue_reference_v<Type>) {
48  using base_type = std::remove_reference_t<Type>;
49 
50  switch(op) {
51  case operation::REF:
52  case operation::CREF:
53  as_any(to).vtable = (op == operation::REF) ? basic_vtable<Type> : basic_vtable<const base_type &>;
54  [[fallthrough]];
55  case operation::COPY:
56  case operation::MOVE:
57  as_any(to).instance = from.instance;
58  [[fallthrough]];
59  case operation::DTOR:
60  break;
61  case operation::COMP:
62  return compare<std::remove_const_t<base_type>>(from.instance, to) ? to : nullptr;
63  case operation::ADDR:
64  return std::is_const_v<base_type> ? nullptr : from.instance;
65  case operation::CADDR:
66  return from.instance;
67  case operation::TYPE:
68  as_type_info(to) = type_id<std::remove_const_t<base_type>>();
69  break;
70  }
71  } else if constexpr(in_situ<Type>) {
72  #if __cpp_lib_launder >= 201606L
73  auto *instance = const_cast<Type *>(std::launder(reinterpret_cast<const Type *>(&from.storage)));
74  #else
75  auto *instance = const_cast<Type *>(reinterpret_cast<const Type *>(&from.storage));
76  #endif
77 
78  switch(op) {
79  case operation::COPY:
80  new (&as_any(to).storage) Type{std::as_const(*instance)};
81  break;
82  case operation::MOVE:
83  new (&as_any(to).storage) Type{std::move(*instance)};
84  [[fallthrough]];
85  case operation::DTOR:
86  instance->~Type();
87  break;
88  case operation::COMP:
89  return compare<Type>(instance, to) ? to : nullptr;
90  case operation::ADDR:
91  case operation::CADDR:
92  return instance;
93  case operation::REF:
94  case operation::CREF:
95  as_any(to).vtable = (op == operation::REF) ? basic_vtable<Type &> : basic_vtable<const Type &>;
96  as_any(to).instance = instance;
97  break;
98  case operation::TYPE:
99  as_type_info(to) = type_id<Type>();
100  break;
101  }
102  } else {
103  switch(op) {
104  case operation::COPY:
105  as_any(to).instance = new Type{*static_cast<const Type *>(from.instance)};
106  break;
107  case operation::REF:
108  case operation::CREF:
109  as_any(to).vtable = (op == operation::REF) ? basic_vtable<Type &> : basic_vtable<const Type &>;
110  [[fallthrough]];
111  case operation::MOVE:
112  as_any(to).instance = from.instance;
113  break;
114  case operation::DTOR:
115  delete static_cast<const Type *>(from.instance);
116  break;
117  case operation::COMP:
118  return compare<Type>(from.instance, to) ? to : nullptr;
119  case operation::ADDR:
120  case operation::CADDR:
121  return from.instance;
122  case operation::TYPE:
123  as_type_info(to) = type_id<Type>();
124  break;
125  }
126  }
127  }
128 
129  return nullptr;
130  }
