85 bool setter([[maybe_unused]] meta_handle handle, [[maybe_unused]] meta_any index, [[maybe_unused]] meta_any value) {
86  bool accepted = false;
87 
88  if constexpr(!Const) {
89  if constexpr(std::is_function_v<std::remove_pointer_t<decltype(Data)>> || std::is_member_function_pointer_v<decltype(Data)>) {
90  using helper_type = meta_function_helper_t<decltype(Data)>;
91  using data_type = std::tuple_element_t<!std::is_member_function_pointer_v<decltype(Data)>, typename helper_type::args_type>;
92  static_assert(std::is_invocable_v<decltype(Data), Type &, data_type>);
93  auto *direct = value.try_cast<data_type>();
94  auto *clazz = handle.data<Type>();
95 
96  if(clazz && (direct || value.convert<data_type>())) {
97  std::invoke(Data, *clazz, direct ? *direct : value.cast<data_type>());
98  accepted = true;
99  }
100  } else if constexpr(std::is_member_object_pointer_v<decltype(Data)>) {
101  using data_type = std::remove_cv_t<std::remove_reference_t<decltype(std::declval<Type>().*Data)>>;
102  static_assert(std::is_invocable_v<decltype(Data), Type *>);
103  auto *clazz = handle.data<Type>();
104 
105  if constexpr(std::is_array_v<data_type>) {
106  using underlying_type = std::remove_extent_t<data_type>;
107  auto *direct = value.try_cast<underlying_type>();
108  auto *idx = index.try_cast<std::size_t>();
109 
110  if(clazz && idx && (direct || value.convert<underlying_type>())) {
111  std::invoke(Data, clazz)[*idx] = direct ? *direct : value.cast<underlying_type>();
112  accepted = true;
113  }
114  } else {
115  auto *direct = value.try_cast<data_type>();
116 
117  if(clazz && (direct || value.convert<data_type>())) {
118  std::invoke(Data, clazz) = (direct ? *direct : value.cast<data_type>());
119  accepted = true;
120  }
121  }
122  } else {
123  static_assert(std::is_pointer_v<decltype(Data)>);
124  using data_type = std::remove_cv_t<std::remove_reference_t<decltype(*Data)>>;
125 
126  if constexpr(std::is_array_v<data_type>) {
127  using underlying_type = std::remove_extent_t<data_type>;
128  auto *direct = value.try_cast<underlying_type>();
129  auto *idx = index.try_cast<std::size_t>();
130 
131  if(idx && (direct || value.convert<underlying_type>())) {
132  (*Data)[*idx] = (direct ? *direct : value.cast<underlying_type>());
133  accepted = true;
134  }
135  } else {

137

139  *Data = (direct ? *direct : value.cast<data_type>());
140  accepted = true;

150 template<typename Type, auto Data, typename Policy>
151 meta_any getter([[maybe_unused]] meta_handle handle, [[maybe_unused]] meta_any index) {
152  auto dispatch = [](auto &&value) {
153  if constexpr(std::is_same_v<Policy, as_void_t>) {
154  return meta_any{std::in_place_type<void>};
155  } else if constexpr(std::is_same_v<Policy, as_alias_t>) {
156  return meta_any{as_alias, std::forward<decltype(value)>(value)};
157  } else {
158  static_assert(std::is_same_v<Policy, as_is_t>);
159  return meta_any{std::forward<decltype(value)>(value)};
160  }
161  };
162 
163  if constexpr(std::is_function_v<std::remove_pointer_t<decltype(Data)>> || std::is_member_function_pointer_v<decltype(Data)>) {
164  static_assert(std::is_invocable_v<decltype(Data), Type &>);
165  auto *clazz = handle.data<Type>();
166  return clazz ? dispatch(std::invoke(Data, *clazz)) : meta_any{};
167  } else if constexpr(std::is_member_object_pointer_v<decltype(Data)>) {
168  using data_type = std::remove_cv_t<std::remove_reference_t<decltype(std::declval<Type>().*Data)>>;
169  static_assert(std::is_invocable_v<decltype(Data), Type *>);
170  auto *clazz = handle.data<Type>();
171 
172  if constexpr(std::is_array_v<data_type>) {
173  auto *idx = index.try_cast<std::size_t>();
174  return (clazz && idx) ? dispatch(std::invoke(Data, clazz)[*idx]) : meta_any{};
175  } else {
176  return clazz ? dispatch(std::invoke(Data, clazz)) : meta_any{};
177  }
178  } else {
179  static_assert(std::is_pointer_v<std::decay_t<decltype(Data)>>);
180 
181  if constexpr(std::is_array_v<std::remove_pointer_t<decltype(Data)>>) {
182  auto *idx = index.try_cast<std::size_t>();
183  return idx ? dispatch((*Data)[*idx]) : meta_any{};
184  } else {
185  return dispatch(*Data);
186  }
187  }
188 }

191 template<typename Type, auto Candidate, typename Policy, std::size_t... Indexes>
192 meta_any invoke([[maybe_unused]] meta_handle handle, meta_any *args, std::index_sequence<Indexes...>) {
193  using helper_type = meta_function_helper_t<decltype(Candidate)>;
194 
195  auto dispatch = [](auto *... args) {
196  if constexpr(std::is_void_v<typename helper_type::return_type> || std::is_same_v<Policy, as_void_t>) {
197  std::invoke(Candidate, *args...);
198  return meta_any{std::in_place_type<void>};
199  } else if constexpr(std::is_same_v<Policy, as_alias_t>) {
200  return meta_any{as_alias, std::invoke(Candidate, *args...)};
201  } else {
202  static_assert(std::is_same_v<Policy, as_is_t>);
203  return meta_any{std::invoke(Candidate, *args...)};
204  }
205  };
206 
207  [[maybe_unused]] const auto direct = std::make_tuple([](meta_any *any, auto *instance) {
208  using arg_type = std::remove_reference_t<decltype(*instance)>;
209 
210  if(!instance && any->convert<arg_type>()) {
211  instance = any->try_cast<arg_type>();
212  }
213 
214  return instance;
215  }(args+Indexes, (args+Indexes)->try_cast<std::tuple_element_t<Indexes, typename helper_type::args_type>>())...);
216 
217  if constexpr(std::is_function_v<std::remove_pointer_t<decltype(Candidate)>>) {
218  return (std::get<Indexes>(direct) && ...) ? dispatch(std::get<Indexes>(direct)...) : meta_any{};
219  } else {
220  auto *clazz = handle.data<Type>();
221  return (clazz && (std::get<Indexes>(direct) && ...)) ? dispatch(clazz, std::get<Indexes>(direct)...) : meta_any{};
