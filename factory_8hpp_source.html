154 template<typename Type, auto Data, typename Policy>
155 meta_any getter([[maybe_unused]] meta_handle handle, [[maybe_unused]] meta_any index) {
156  auto dispatch = [](auto &&value) {
157  if constexpr(std::is_same_v<Policy, as_void_t>) {
158  return meta_any{std::in_place_type<void>};
159  } else if constexpr(std::is_same_v<Policy, as_alias_t>) {
160  return meta_any{std::ref(std::forward<decltype(value)>(value))};
161  } else {
162  static_assert(std::is_same_v<Policy, as_is_t>);
163  return meta_any{std::forward<decltype(value)>(value)};
164  }
165  };
166 
167  if constexpr(std::is_function_v<std::remove_reference_t<std::remove_pointer_t<decltype(Data)>>> || std::is_member_function_pointer_v<decltype(Data)>) {
168  static_assert(std::is_invocable_v<decltype(Data), Type &>);
169  auto * const clazz = meta_any{handle}.try_cast<Type>();
170  return clazz ? dispatch(std::invoke(Data, *clazz)) : meta_any{};
171  } else if constexpr(std::is_member_object_pointer_v<decltype(Data)>) {
172  using data_type = std::remove_cv_t<std::remove_reference_t<decltype(std::declval<Type>().*Data)>>;
173  static_assert(std::is_invocable_v<decltype(Data), Type *>);
174  auto * const clazz = meta_any{handle}.try_cast<Type>();
175 
176  if constexpr(std::is_array_v<data_type>) {
177  auto * const idx = index.try_cast<std::size_t>();
178  return (clazz && idx) ? dispatch(std::invoke(Data, clazz)[*idx]) : meta_any{};
179  } else {
180  return clazz ? dispatch(std::invoke(Data, clazz)) : meta_any{};
181  }
182  } else {
183  static_assert(std::is_pointer_v<std::decay_t<decltype(Data)>>);
184 
185  if constexpr(std::is_array_v<std::remove_pointer_t<decltype(Data)>>) {
186  auto * const idx = index.try_cast<std::size_t>();
187  return idx ? dispatch((*Data)[*idx]) : meta_any{};
188  } else {
189  return dispatch(*Data);
190  }
191  }
192 }

204  return meta_any{std::ref(std::invoke(Candidate, *params...))};
205  } else {

207  return meta_any{std::invoke(Candidate, *params...)};
208  }

212  using arg_type = std::remove_reference_t<decltype(*instance)>;
